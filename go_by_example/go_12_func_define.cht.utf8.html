函數是Go語言的重要內容。

package main

import "fmt"

// 這個函數計算兩個int型輸入數據的和，並返回int型的和
func plus(a int, b int) int {
    // Go需要使用return語句顯式地返回值
    return a + b
}

func main() {
    // 函數的調用方式很簡單
    // "名稱(參數列表)"
    res := plus(1, 2)
    fmt.Println("1+2 =", res)
}

輸出結果為

1+2 = 3
