for循環是Go語言唯一的循環結構。這裡有三個基本的for循環類型。

package main

import "fmt"

func main() {

    // 最基本的一種，單一條件循環
    // 這個可以代替其他語言的while循環
    i := 1
    for i <= 3 {
        fmt.Println(i)
        i = i + 1
    }

    // 經典的循環條件初始化/條件判斷/循環後條件變化
    for j := 7; j <= 9; j++ {
        fmt.Println(j)
    }

    // 無條件的for循環是死循環，除非你使用break跳出循環或者
    // 使用return從函數返回
    for {
        fmt.Println("loop")
        break
    }
}

輸出結果

1
2
3
7
8
9
loop
