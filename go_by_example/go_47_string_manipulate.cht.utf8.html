strings 標准庫提供了很多字符串操作相關的函數。這裡提供的幾個例子是讓你先對這個包有個基本了解。

package main

import s "strings"
import "fmt"

// 這裡給fmt.Println起個別名，因為下面我們會多處使用。
var p = fmt.Println

func main() {

    // 下面是strings包裡面提供的一些函數實例。注意這裡的函數並不是
    // string對象所擁有的方法，這就是說使用這些字符串操作函數的時候
    // 你必須將字符串對象作為第一個參數傳遞進去。
    p("Contains:  ", s.Contains("test", "es"))
    p("Count:     ", s.Count("test", "t"))
    p("HasPrefix: ", s.HasPrefix("test", "te"))
    p("HasSuffix: ", s.HasSuffix("test", "st"))
    p("Index:     ", s.Index("test", "e"))
    p("Join:      ", s.Join([]string{"a", "b"}, "-"))
    p("Repeat:    ", s.Repeat("a", 5))
    p("Replace:   ", s.Replace("foo", "o", "0", -1))
    p("Replace:   ", s.Replace("foo", "o", "0", 1))
    p("Split:     ", s.Split("a-b-c-d-e", "-"))
    p("ToLower:   ", s.ToLower("TEST"))
    p("ToUpper:   ", s.ToUpper("test"))
    p()

    // 你可以在strings包裡面找到更多的函數

    // 這裡還有兩個字符串操作方法，它們雖然不是strings包裡面的函數，
    // 但是還是值得提一下。一個是獲取字符串長度，另外一個是從字符串中
    // 獲取指定索引的字符
    p("Len: ", len("hello"))
    p("Char:", "hello"[1])
}
